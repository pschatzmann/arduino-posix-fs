758 static char* itoa(int n, char s[]) {
759 int i, sign;
760 if ((sign = n) < 0) /* record sign */
761 n = -n; /* make n positive */
762 i = 0;
763 do { /* generate digits in reverse order */
764 s[i++] = n % 10 + '0'; /* get next digit */
765 } while ((n /= 10) > 0); /* delete it */
766 if (sign < 0)
767 s[i++] = '-';
768 s[i] = '\0';
769 reverse(s);
770 return s;
771 }
