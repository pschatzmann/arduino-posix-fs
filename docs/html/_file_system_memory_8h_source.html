221 ssize_t read(int fd, void *data, size_t size) override {
222 FS_LOGI("read: fd='%d' size=%d", fd, (int)size);
223 if (size == 0) {
224 return 0;
225 }
226 // If we did not find any content we return 0
227 RegEntry &entry = DefaultRegistry.getEntry(fd);
228 RegContentMemory *p_memory = getContent(entry);
229 if (p_memory == nullptr) {
230 FS_LOGW("No content for %s", entry.file_name);
231 return 0;
232 }
233 // If we are at the end we return 0
234 size_t pos = p_memory->current_pos;
235 if (pos >= p_memory->size) {
236 FS_LOGD("=> read: pos=%d file-size=%d fd=%d -> %d", pos, p_memory->size,
237 fd, 0);
238 return 0;
239 }
240 // Copy requested data
241 int len = min(size, p_memory->size - pos);
242 p_memory->current_pos += len;
243 memmove(data, p_memory->data + pos, len);
244 FS_LOGD("=> read: pos=%d size=%d fd=%d -> %d", pos, (int)size, fd, len);
245 return len;
246 }

295 off_t lseek(int fd, off_t offset, int whence) override {
296 FS_LOGI("lseek: fd='%%' ", fd);
297 RegEntry &entry = DefaultRegistry.getEntry(fd);
298 RegContentMemory *p_memory = getContent(entry);
299 if (p_memory == nullptr) {
300 return -1;
301 }
302 switch (whence) {
303 case SEEK_SET:
304 p_memory->current_pos = offset;
305 if (p_memory->current_pos > p_memory->size) {
306 int diff = p_memory->size - p_memory->current_pos;
307 offset -= diff;
308 p_memory->current_pos = p_memory->size - 1;
309 }
310 break;
311 case SEEK_CUR:
312 p_memory->current_pos += offset;
313 if (p_memory->current_pos > p_memory->size) {
314 int diff = p_memory->size - p_memory->current_pos;
315 offset -= diff;
316 p_memory->current_pos = p_memory->size - 1;
317 }
318 break;
319 case SEEK_END:
320 long pos = p_memory->size - offset;
321 if (pos < 0) {
322 p_memory->current_pos = 0;
323 offset = offset + p_memory->current_pos;
324 } else {
325 p_memory->current_pos = pos;
326 }
327 break;
328 }
329 return offset;
330 }

355 dirent *readdir(DIR *dir) override {
356 FS_TRACEI();
357 DIR_EXT *p_dir = (DIR_EXT *)dir;
358 if (p_dir->pos >= p_dir->files.size()) {
359 FS_LOGD("==> readdir: pos=%d size=%d END", p_dir->pos,
360 p_dir->files.size());
361 p_dir->pos++;
362 return nullptr;
363 }
364 RegEntry *entry = p_dir->files[p_dir->pos];
365 p_dir->pos++;
366
367 // we return only the file name w/o path
368 int len = strlen(p_dir->dir);
369 if (entry->file_name[len] == '/') {
370 len++;
371 }
372 const char *result_name = (entry->file_name) + len;
373
374 // copy filname to dirent
375 strncpy(p_dir->actual_dirent.d_name, result_name, MAXNAMLEN);
376 FS_LOGD("==> readdir: pos=%d size=%d %s", p_dir->pos, p_dir->files.size(),
377 result_name);
