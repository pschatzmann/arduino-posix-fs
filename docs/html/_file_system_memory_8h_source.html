169 ssize_t read(int fd, void *data, size_t size) override {
170 FS_TRACED();

172 return 0;

174 RegEntry &entry = DefaultRegistry.getEntry(fd);
175 RegContentMemory *p_memory = getContent(entry);
176 if (p_memory == nullptr) {
177 FS_LOGW("No content for %s", entry.file_name);

180 int pos = p_memory->current_pos;
181 int len = min(size, p_memory->size - pos);
182 p_memory->current_pos += len;
183 memmove(data, p_memory->data + pos, len);
184 FS_LOGD("=> read: pos=%d size=%d fd=%d -> %d", pos, (int)size, fd, len);

209 off_t lseek(int fd, off_t offset, int whence) override {
210 FS_TRACED();
211 RegEntry &entry = DefaultRegistry.getEntry(fd);
212 RegContentMemory *p_memory = getContent(entry);
213 if (p_memory == nullptr) {
214 return -1;
215 }
216 switch (whence) {
217 case SEEK_SET:
218 p_memory->current_pos = offset;
219 if (p_memory->current_pos > p_memory->size) {
220 int diff = p_memory->size - p_memory->current_pos;
221 offset -= diff;
222 p_memory->current_pos = p_memory->size - 1;
223 }
224 break;
225 case SEEK_CUR:
226 p_memory->current_pos += offset;
227 if (p_memory->current_pos > p_memory->size) {
228 int diff = p_memory->size - p_memory->current_pos;
229 offset -= diff;
230 p_memory->current_pos = p_memory->size - 1;
231 }
232 break;
233 case SEEK_END:
234 p_memory->current_pos = p_memory->size - offset;
235 if (p_memory->current_pos < 0) {
236 p_memory->current_pos = 0;
237 offset = offset + p_memory->current_pos;
238 }
239 break;
240 }
241 return offset;
242 }
